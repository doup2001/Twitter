create database twit;
use twit;

CREATE TABLE account (
    firstname VARCHAR(255),
    lastname VARCHAR(255),
    id VARCHAR(20) PRIMARY KEY,
    password VARCHAR(255)
    birthdate DATE,
    email VARCHAR(255),
    gender VARCHAR(10);
);

CREATE TABLE follow (
    fromUser VARCHAR(20),
    toUser VARCHAR(20),
    PRIMARY KEY(fromUser, toUser),
    FOREIGN KEY(fromUser) REFERENCES account(id),
    FOREIGN KEY(toUser) REFERENCES account(id)
);	

CREATE TABLE article (
    num INT,
    ID VARCHAR(20),
    article TEXT,
    createdAt TIMESTAMP,
    PRIMARY KEY(num),
    FOREIGN KEY(ID) REFERENCES account(id)
);

CREATE TABLE like_post (
    post_num INT,
    liker_id VARCHAR(20),
    PRIMARY KEY(post_num, liker_id),
    FOREIGN KEY(post_num) REFERENCES article(num),
    FOREIGN KEY(liker_id) REFERENCES account(id)
);

select * from article;
select * from follow;
select * from account;
